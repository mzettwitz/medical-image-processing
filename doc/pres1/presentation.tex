\subsection{Probleme}
\begin{frame}
\frametitle{Probleme}
\begin{itemize}
\item Nadel strahlt
\item Rauschen in Bildern
\item Nadel ist teilweise nicht in Schicht sichtbar
\item Nadelspitze unterbrochen
\item Knochen und Nadel haben ähnliche Hounsfield Werte
\end{itemize}
\end{frame}
